/**
 * Tipo: controller Nombre: AdministrarFechaVencimientoControlador
 * Descripción: Es el controlador del modulo
 *
 * @autor Mauro Xavier Rivera Rasury (mjrr210316)
 * @copyright SRI 2016
 * @param $scope - Ambito del controlador
 * @param $translate - Ambito de traduccion
 
 */
(function (angular) {
    'use strict';
 
    angular.module('gestor').controller('GestorControlador',GestorControlador);
 
    GestorControlador.$inject = ['$scope', '$translate'];
 
    function GestorControlador ($scope,  $translate) {
       $scope.alerts = [];
 
       /**
         * @author Mauro Xavier Rivera Rasury (mjrr210316)
         * @description Funcion para saludar
         * @return void
         */
       $scope.saludar = function () {
           $scope.alerts = [];
       };
 
       /**
         * @author Mauro Xavier Rivera Rasury (mjrr210316)
         * @description Funcion para limpiar las alertas
         * @return void
         */
       $scope.limpiar = function () {
           $scope.alerts = [];
       };
 
       /**
         * @author Mauro Xavier Rivera Rasury (mjrr210316)
         * @description Funcion para cerrar la alerta
         * @return void
         */
       $scope.closeAlert = function (index) {
           $scope.alerts.splice(index, 1);
       };
 
       /**
         * @author Mauro Xavier Rivera Rasury (mjrr210316)
         * @description Funcion para cambiar la alerta
         * @return void
         */
       $scope.switchLanguage = function(key) {
           //console.log('cambia lenguaje a:'+key);
           $scope.lenguajeActual = key;
           $translate.use(key);
       };
    }
}(window.angular));
