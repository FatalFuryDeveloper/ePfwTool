        vm.consultarMetodos = function(datos){

                var datos = {
                    metodo: "POST",

                    data:  {
                        funcion: vm.datos.funcion
                    }

                //Se consume el servicio pasando los datos correspondientes
                ComunServicio.invocarPeticion(datos).then(function (respuesta) {

                    promesa.reject(error);
                });
