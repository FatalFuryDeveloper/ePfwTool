 * @type    Servicio
 * @name    InicioSesionServicio
 * @desc    Es el servicio para iniciar sesion en el sistema
 * @autor   Mauro Xavier Rivera Rasury (fatalfurydeveloper)

 * @version 1.0
 */
(function(angular){
    'use strict';
 
    angular.module('gestor').service('InicioSesionServicio', InicioSesionServicio);
    angular.module('sri').service('InicioSesionServicio', InicioSesionServicio);
 
    InicioSesionServicio.$inject = ['$http', '$q', 'ComunServicio'];
 
    function InicioSesionServicio($http, $q, ComunServicio) {
        var vm = this ;
 
        /**
         * @description Funcion para realizar peticiones HTTP
         * @param {datos} [Objeto] [Datos a consultar]
         * @return {[diferido.promise]} [Devuelve la promesa del servicio]
         */
        vm.login = function(usuario){
            var promesa = $q.defer();
 
            //Se Obtiene la Url y se arma la trama correspondiente del servicio
            ComunServicio.obtenerPropiedades().then(function(resp) {
                //var clave = CryptoJS.SHA512(usuario.password);
                //var clave = usuario.password;
                var datos = {
                    metodo: "POST",
                    url:    resp.data.rcs.php.ruta + resp.data.rcs.php.login,
                    data:  {
                        usuario: usuario.email,
                        password: usuario.id,
                    }
                };
                //Se consume el servicio pasando los datos correspondientes
                ComunServicio.invocarPeticion(datos).then(function (respuesta) {
                    //SesionesServicio.set(datos.data.usuario, true);
                    promesa.resolve(respuesta);
                }, function (error) {
                    promesa.reject(error);
                });
            });
            return promesa.promise;
        };
 
        /**
         * @description Funcion para realizar peticiones HTTP
         * @param {datos} [Objeto] [Datos a consultar]
         * @return {[diferido.promise]} [Devuelve la promesa del servicio]
         */

           // SesionesServicio.clear();
            //SesionesServicio.setList("mensaje", {texto:"¡Hasta pronto!", tipo:"success"});
            //$location.path("/login");
            return ;
        };
    }
}(window.angular));
