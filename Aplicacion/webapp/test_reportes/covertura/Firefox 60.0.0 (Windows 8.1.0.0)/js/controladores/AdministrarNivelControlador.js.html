/**
 * Tipo: controller
 * Nombre: AdministrarNivelControlador
 * @autor Mauro Xavier Rivera Rasury (mjrr210316)

(function(angular) {
 	'use strict';
 	angular.module('sri').controller('AdministrarNivelControlador',AdministrarNivelControlador);
 
 	AdministrarNivelControlador.$inject = ['$rootScope', '$scope',  '$uibModal', '$state', '$translate', '$timeout',
        'I18nFactory', '$http', '$q', '$compile', 'DTOptionsBuilder', 'DTColumnBuilder', 'NivelServicio'];
 
 	function AdministrarNivelControlador($rootScope, $scope, $uibModal, $state, $translate, $timeout,
        i18nFactory, $http,  $q, $compile, DTOptionsBuilder, DTColumnBuilder, NivelServicio) {
 
 		/** ********** DECLARACION DE VARIABLES ************ */
 		var vm = this;
 		// Variables para instanciar los campos de los formularios
        vm.expRegular = /^[0-9\/]{0,10}$/;
        vm.expRegularFinal = "";
 		vm.alerts = [];
        vm.guardar = true;
        vm.desactivarBtnAceptar = true;
        vm.txtNivel = "";
        vm.txtDescripcion = "";
        vm.cmbEstado = null;
        vm.cmbDsEstado = [{ codigo: 0, estado: "Inactivo" }, { codigo: 1, estado: "Activo" }];
        // Variables para instanciar las tablas de la aplicacion
        vm.dtAdministrar = [];
        vm.datosNiveles = [];
        // Variables para llamar a funciones
        vm.aceptar = aceptar;
        vm.consultar = consultarNivel;
        vm.crearFilas = crearFilas;
        vm.agregarAlerta = agregarAlerta;
        vm.init = inicializar;
 		vm.limpiarFiltro = borrarFiltros;
 		/** ********** FUNCIONES ************ */
 		$rootScope.$on('$translateChangeSuccess', inicializar);
 
        /**
         * @author Mauro Xavier Rivera Rasury (mjrr210316)
         * @description Funcion para inicializar datos en nuestra aplicación
         * @return void
         */
        function inicializar(){
            vm.idioma = i18nFactory.idioma();
            $timeout(consultarNivel, 200);
            cargarCombos();
            tablaAdministracionNivel();
        }
 
        /**
         * @author Mauro Xavier Rivera Rasury (mjrr210316)
         * @description validacion de campos obligatorios para activar boton agregar
         * @returns void
         */
        vm.validar = function (){
            if (vm.txtNivel !== "" && vm.txtDescripcion !== "" && vm.cmbEstado !== null) {
                vm.desactivarBtnAceptar = false;
            } else {
                vm.desactivarBtnAceptar = true;
            }
        };
 
        /**
         * @author Mauro Xavier Rivera Rasury (mjrr210316)
         * @description Funcion para cargar los combos de la vista
         * @return void
         */
        function cargarCombos() {
            // Variables de opciones de los combos de la aplicacion
            var opcion =  i18nFactory.i18nTraduccion('FRM.SELECCIONEOPCION');
            vm.optionsEstado       = opcionCombo(opcion, "estado", "codigo");
        }
 
        /**
         * @author Mauro Xavier Rivera Rasury (mjrr210316)
         * @description Funcion para instanciar las opciones de los combos
         * @param {string} [tipoCombo] [mensaje de phaceholder]
         * @param {string} [texto] [valor del combo]
         * @param {int} [codigo] [valor del combo]
         * @returns opcionCombo - Opciones del combo
         */
        function opcionCombo(tipoCombo, texto, codigo){
            var opcionCombo = {
                placeholder: tipoCombo,
                dataTextField: texto,
                value: (codigo, texto),
                filter: "contains",
                suggest: true,
                valuePrimitive: true,
                autoBind: false
            };
            return opcionCombo;
        }
 
        /**
         * @author Mauro Xavier Rivera Rasury (mjrr210316)
         * @description Funcion para inicializar tabla
         * @return void
         */
        function tablaAdministracionNivel(){
            // Variable inicializada para las opciones tabla obligacion
            vm.dtAdministrarOptions = DTOptionsBuilder.fromFnPromise(promesaAdministrar)
                .withPaginationType('simple_numbers')
                .withOption('info', true)
                .withOption('bFilter', true)
                .withOption('fnDrawCallback', devolverLlamada)
                .withOption('sDom', '<"top"lif>rt<"bottom"p><"clear">')
                .withOption('createdRow', crearFilas)
                .withOption('order', [0, 'asc'])
                .withLanguageSource('recursos/lenguajes/'+vm.idioma+'.json');
            // Variable inicializada para las columnas de la tabla obligacion
            vm.dtAdministrarColumns = [
                DTColumnBuilder.newColumn('niv_id')
                    .withTitle($translate('TBL.ID'))
                    .withClass('text-center')
                    .withOption('width', '10%'),
                DTColumnBuilder.newColumn("niv_nombre")
                    .withTitle($translate('TBL.NIVEL'))
                    .withOption('width', '30%'),
                DTColumnBuilder.newColumn("niv_descripcion")
                    .withTitle($translate('TBL.DESCRIPCION'))
                    .withOption('width', '40%'),
                DTColumnBuilder.newColumn("niv_estado")
                    .withTitle($translate('TBL.ESTADO'))
                    .withOption('width', '10%'),
                DTColumnBuilder.newColumn(null)
                    .withTitle($translate('TBL.ACCIONES'))
                    .withOption('width', '10%')
                    .withClass('text-center')
                    .renderWith(accionesHtml)
             ];
        }
 
        /**
         * @author Mauro Xavier Rivera Rasury (mjrr210316)
         * @description Funcion para crear acciones de filas en tabla administracion fecha vencimiento
         * @returns {Objeto} [accionesHtml] [acciones de la tabla]
         */
        function accionesHtml(data) {
            return  '<a class="btn" data-ng-if="\'' + data.niv_estado + '\' === \'Activo\'" ' +
                    'data-ng-click="admnivel.cambiarEstado(' + data.niv_id +',\''+ data.niv_estado +'\')">' +
                    '<i class="glyphicon glyphicon-ban-circle  icono-celeste" data-uib-tooltip="Inactivar" ' +
                    'data-tooltip-placement="top" tooltip-class="mensaje-ayuda-contextual""></i></a>' +
                    '<a class="btn" data-ng-if="\'' + data.niv_estado + '\' === \'Inactivo\'" ' +
                    'data-ng-click="admnivel.cambiarEstado(' + data.niv_id +',\''+ data.niv_estado +'\')">' +
                    '<i class="glyphicon glyphicon-ok-circle  icono-celeste" data-uib-tooltip="Activar" ' +
                    'data-tooltip-placement="top" tooltip-class="mensaje-ayuda-contextual""></i></a>' +
                    '<a class="btn" data-ng-click="admnivel.editar('+data.niv_id+',\''+data.niv_nombre+'\','+
                    '\''+data.niv_descripcion+'\',\''+data.niv_estado+'\')">' +
                    '<i class="icon-editar icono-celeste" data-uib-tooltip="Editar" ' +
                    'data-tooltip-placement="top" tooltip-class="mensaje-ayuda-contextual"></i></a>' +
                    '<a class="btn" data-ng-click="admnivel.eliminar('+data.niv_id+')">'+
                    '<i class="icon-eliminar  icono-rojo " data-uib-tooltip="Eliminar" ' +
                    'data-tooltip-placement="top" tooltip-class="mensaje-ayuda-contextual"></i></a>';
        }
 
        /**
         * @author Mauro Xavier Rivera Rasury (mjrr210316)
         * @description Funcion para armar elementos de la tabla
         * @return void
         */
        function devolverLlamada() {
            $compile(angular.element("#tblAdministrarNivel" + "_length").contents())($scope);
        }
 
        /**
         * @author Mauro Xavier Rivera Rasury (mjrr210316)
         * @description Promesa para la tabla de informacion sujeto;
         * @returns deferred.promise Promesa
         */
        function promesaAdministrar() {
            var deferred = $q.defer();
                deferred.resolve(vm.datosNiveles);
            return deferred.promise;
        }
 
        /**
         * @author Mauro Xavier Rivera Rasury (mjrr210316)
         * @description Funcion para crear filas de tabla
         * @params row columnas de la tabla
         * @returns void
         */
        function crearFilas(row) {
            //crea filas necesarias para tabla
            $compile(angular.element(row).contents())($scope);
        }
 
        function aceptar(){
            if(vm.guardar === true){
                insertarNivel();
            }else{
                editarNivel();
            }
        }
 
        vm.editar = function(id,nombre,descripcion,estado){
            console.log(id,nombre,descripcion,estado);
            vm.alerts = [];
            vm.guardar = false;
            vm.id = id;
            vm.txtNivel = nombre;
            vm.txtDescripcion = descripcion;
            vm.cmbEstado = {codigo:0,estado:estado};
            vm.existeId= false;
        };
 
        vm.cambiarEstado = function(id,estado){
            vm.alerts = [];
            vm.id = id;
            if(estado === "Activo"){estado="Inactivo";}else{estado="Activo";}
            vm.cmbEstado = {estado:estado};
            cambiarEstadoNivel();
        };
 
        vm.eliminar = function(id){
            vm.id = id;
            abrirModal();
        };
 
 
        function consultarNivel(){
            var datos= {
                    funcion:"consultarNiveles"
            };
            var promesa = NivelServicio.consultarNiveles(datos);
                promesa.then(function (respuesta) {
                    vm.datosNiveles = respuesta.data;
                    vm.dtAdministrar.reloadData();
                },
                function() {
                    agregarAlerta("danger","MSJ.CON.ERROR");
                });
        }
 
        function insertarNivel(){
            var datos= {
                funcion         : "insertarNivel",
                nombre          : vm.txtNivel,
                descripcion     : vm.txtDescripcion,
                estado          : vm.cmbEstado.estado
            };
            var promesa = NivelServicio.insertarNivel(datos);
                promesa.then(function () {
                    borrarFiltros();
                    agregarAlerta("success","MSJ.INS.SUCCESS");
                    consultarNivel();
                },
                function() {
                    agregarAlerta("danger","MSJ.INS.ERROR");
                });
        }
 
        function editarNivel(){
            var datos= {
                funcion         : "modificarNivel",
                id              : vm.id,
                nombre          : vm.txtNivel,
                descripcion     : vm.txtDescripcion,
                estado          : vm.cmbEstado.estado
            };
            var promesa = NivelServicio.editarNivel(datos);
                promesa.then(function () {
                    borrarFiltros();
                    agregarAlerta("success","MSJ.ACT.SUCCESS");
                    consultarNivel();
                },
                function() {
                    agregarAlerta("danger","MSJ.ACT.ERROR");
                });
        }
 
        function cambiarEstadoNivel(){
            var datos= {
                funcion     :"cambiarEstadoNivel",
                id          : vm.id,
                estado      : vm.cmbEstado.estado
            };
            var promesa = NivelServicio.cambiarEstadoNivel(datos);
                promesa.then(function () {
                    borrarFiltros();
                    agregarAlerta("success","MSJ.ACT.SUCCESS");
                    consultarNivel();
                },
                function() {
                    agregarAlerta("danger","MSJ.ACT.ERROR");
                });
        }
 
         function eliminarNivel(){
            var datos= {
                funcion:"eliminarNivel",
                id: vm.id,
                eliminado: true
            };
            var promesa = NivelServicio.eliminarNivel(datos);
                promesa.then(function () {
                    borrarFiltros();
                    agregarAlerta("success","MSJ.ELI.SUCCESS");
                    consultarNivel();
                },
                function() {
                    agregarAlerta("danger","MSJ.ELI.ERROR");
                });
        }
 
         /**
         * @author Mauro Xavier Rivera Rasury (mjrr210316)
         * @description Funcion para abrir modal de confirmacion
         * @returns void
         */
        function abrirModal() {
            var modalInstance = $uibModal.open({
                animation: 'true',
                templateUrl: 'recursos/paginas/confirmacion-modal.html',//ruta html de modal
                controller: 'ModalControlador',// controlador modal
                controllerAs: 'modalctrl',// Alias controlador modal
                resolve: { //Transfiere al controlador
                    mensaje: function () {
                        return $translate("MSJ.MOD.CONFIRMAR");
                    }
                }
            });
            //Instancia modal Aceptar o cancelar
            modalInstance.result.then(function () {
                eliminarNivel();
               //agregarAlerta("success", "MSJ.ELI.SUCCESS");
            },
            function () {
                agregarAlerta("warning", "MSJ.MOD.CANCELAR");
            });
        }
 
        /**
         * @author Mauro Xavier Rivera Rasury (mjrr210316)
         * @description Funcion para agregar alertas a mostrar
         * @param {string} [tipoMensaje] [Definicion del tipo de alerta]
         * @param {string} [mensaje] [Definicion del cuerpo de la alerta]
         * @returns void
         */
        function agregarAlerta(tipoMensaje, mensaje) {
            vm.alerts.push({ type: tipoMensaje, msg: mensaje });
        }
 
 		/**
         * @author Mauro Xavier Rivera Rasury (mjrr210316)
         * @description Funcion para borrar formulario
         * @returns void
         */
        function borrarFiltros() {
            vm.alerts = [];
            vm.guardar = true;
            vm.desactivarBtnAceptar = true;
            vm.txtNivel = "";
            vm.txtDescripcion = "";
            vm.cmbEstado = null;
        }
	}
}(window.angular));
