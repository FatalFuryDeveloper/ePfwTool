/**
 * Tipo: controller
 * Nombre: GestionSubirControlador
 * Descripción: Es el controlador que interactúa con el servicio de legado
 * para consultar los estados de cumplimientos
 *
 * @autor Mauro Xavier Rivera Rasury (mjrr210316)
 * @copyright SRI 2016
 * @param $rootScope - Ambito de la aplicación
 * @param $scope - Ambito del controlador
 * @param $state -Ambito para direccionar
 * @param $translate - Ambito de traduccion
 * @param I18nFactory - Ambito para traduccion
 * @param $http - Ambito para acciones http
 * @param $q - Ambito de promesas
 * @param $compile - ambito
 * @param DTOptionsBuilder - ambito
 * @param DTColumnBuilder
 * @param SujetoServicio - Servicios para consultar el Sujeto
 * Copyright 2016 Servicio de Rentas Internas.
 * Todos los derechos reservados.

(function(angular) {
 	'use strict';
 	angular.module('gestor').controller('GestionSubirControlador',
 		GestionSubirControlador);
 
 
 	GestionSubirControlador.$inject = ['$rootScope', '$scope', '$state', '$translate', '$timeout', '$filter',
        'I18nFactory', '$http', '$q', '$compile', 'DTOptionsBuilder', 'DTColumnBuilder','GestorServicio'];
 
 	function GestionSubirControlador($rootScope, $scope, $state, $translate, $timeout, $filter,
        i18nFactory, $http,  $q, $compile, DTOptionsBuilder, DTColumnBuilder, GestorServicio) {
 
 		/** ********** DECLARACION DE VARIABLES ************ */
 		var vm = this;
 		// Variables para instanciar los campos de los formularios
        vm.expRegular = /^[0-9\/]{0,10}$/;
        vm.expRegularFinal = "";
        vm.tooltipAeronave = false;
        vm.tooltipPasajero = false;
        vm.tooltipPasajeroNo = false;
 		vm.alerts = [];
        vm.desactivarBtnAceptar = true;
        vm.existeAeronave = true;
        vm.existePasajero = true;
        // Variables para instanciar las tablas de la aplicacion
        vm.dtAdministrar = [];
        vm.informacionGestion = [];
        // Variables para llamar a funciones
        vm.aceptar = aceptar;
        vm.consultar = consultarGestion;
        vm.crearFilas = crearFilas;
        vm.agregarAlerta = agregarAlerta;
        vm.init = inicializar;
 		vm.limpiarFiltro = borrarFiltros;
 		/** ********** FUNCIONES ************ */
 		$rootScope.$on('$translateChangeSuccess', inicializar);
 
        /**
         * @author Mauro Xavier Rivera Rasury (mjrr210316)
         * @description validacion de campos obligatorios para activar boton agregar
         * @returns void
         */
        vm.validarAeronave = function (){
            vm.alerts = [];
            if (vm.txtAeronave !== "" ){
                consultarAeronaveEspecifica();
                consultarAeronaveCapacidad();
            }
            vm.validar();
        };
 
        /**
         * @author Mauro Xavier Rivera Rasury (mjrr210316)
         * @description validacion de campos obligatorios para activar boton agregar
         * @returns void
         */
        vm.validarPasajero = function (){
            vm.alerts = [];
            vm.tooltipPasajero = false;
            vm.tooltipPasajeroNo = false;
            if (vm.txtPasajero !== "" ){
                consultarPasajeroEspecifica();
            }
            vm.validar();
        };
 
        /**
         * @author Mauro Xavier Rivera Rasury (mjrr210316)
         * @description validacion de campos obligatorios para activar boton agregar
         * @returns void
         */
        vm.validar = function (){
            if (vm.existeAeronave !== false && vm.existePasajero  !== false) {
                vm.desactivarBtnAceptar = false;
            } else { vm.desactivarBtnAceptar = true; }
        };
 
        /**
         * @author Mauro Xavier Rivera Rasury (mjrr210316)
         * @description Funcion para inicializar datos en nuestra aplicación
         * @return void
         */
        function inicializar(){
            vm.idioma = i18nFactory.idioma();
            tablaAdministracionNodriza();
        }
 
        /**
         * @author Mauro Xavier Rivera Rasury (mjrr210316)
         * @description Funcion para inicializar tabla
         * @return void
         */
        function tablaAdministracionNodriza(){
            // Variable inicializada para las opciones tabla obligacion
            vm.dtAdministrarOptions = DTOptionsBuilder.fromFnPromise(promesaAdministrar)
                .withPaginationType('simple_numbers')
                .withOption('info', true)
                .withOption('bFilter', true)
                .withOption('fnDrawCallback', devolverLlamada)
                .withOption('sDom', '<"top"lif>rt<"bottom"p><"clear">')
                .withOption('createdRow', crearFilas)
                .withOption('order', [0, 'asc'])
                .withLanguageSource('recursos/lenguajes/'+vm.idioma+'.json');
            // Variable inicializada para las columnas de la tabla obligacion
            vm.dtAdministrarColumns = [
                DTColumnBuilder.newColumn('ges_id_pas')
                    .withTitle("Id Pasajero")
                    .withOption('width', '20%'),
                DTColumnBuilder.newColumn("pas_nombre")
                    .withTitle("Nombre Pasajero")
                    .withOption('width', '80%')
             ];
        }
 
        /**
         * @author Mauro Xavier Rivera Rasury (mjrr210316)
         * @description Funcion para armar elementos de la tabla
         * @return void
         */
        function devolverLlamada() {
            $compile(angular.element("#tblAdministrarEC" + "_length").contents())($scope);
        }
 
        /**
         * @author Mauro Xavier Rivera Rasury (mjrr210316)
         * @description Promesa para la tabla de informacion sujeto;
         * @returns deferred.promise Promesa
         */
        function promesaAdministrar() {
            var deferred = $q.defer();
                deferred.resolve(vm.informacionGestion);
            return deferred.promise;
        }
 
        /**
         * @author Mauro Xavier Rivera Rasury (mjrr210316)
         * @description Funcion para crear filas de tabla
         * @params row columnas de la tabla
         * @returns void
         */
        function crearFilas(row) {
            //crea filas necesarias para tabla
            $compile(angular.element(row).contents())($scope);
        }
 
        function aceptar(){
            consultarGestionPasajeroAeronave();
        }
 
 
        function consultarAeronaveEspecifica(){
            var datos= {
                metodo: "POST",
                url:"recursos/js/servicios/consultar.php",
                data: {
                    funcion:"consultarAeronaveEspecifica",
                    codigo: vm.txtAeronave
                }
            }
            var promesa = GestorServicio.llamarFuncionPHP(datos);
                promesa.then(function (respuesta) {
                    vm.existe = respuesta.data[0].existe;
                    if(vm.existe === "0"){
                        vm.existeAeronave = false;
                        vm.tooltipAeronave = true;
                    }else{
                        vm.tooltipAeronave = false;
                        vm.existeAeronave = true;
                    }
                },
                function() {
                    agregarAlerta("danger","Error al consultar");
                });
        }
 
        function consultarAeronaveCapacidad(){
            var datos= {
                metodo: "POST",
                url:"recursos/js/servicios/consultar.php",
                data: {
                    funcion:"consultarAeronaveCapacidad",
                    codigo: vm.txtAeronave
                }
            }
            var promesa = GestorServicio.llamarFuncionPHP(datos);
                promesa.then(function (respuesta) {
                    vm.capacidad = respuesta.data[0].aer_capacidad;
                },
                function() {
                    agregarAlerta("danger","Error al consultar");
                });
        }
 
        function consultarPasajeroEspecifica(){
            var datos= {
                metodo: "POST",
                url:"recursos/js/servicios/consultar.php",
                data: {
                    funcion:"consultarPasajeroEspecifica",
                    codigo: vm.txtPasajero
                }
            }
            var promesa = GestorServicio.llamarFuncionPHP(datos);
                promesa.then(function (respuesta) {
                    vm.existe = respuesta.data[0].existe;
                    if(vm.existe === "0"){
                        vm.existePasajero = false;
                        vm.tooltipPasajero = true;
                    }else{
                        vm.tooltipPasajero = false;
                        vm.existePasajero = true;
                    }
                },
                function() {
                    agregarAlerta("danger","Error al consultar");
                });
        }
 
        function consultarGestionPasajeroAeronave(){
            var datos= {
                metodo: "POST",
                url:"recursos/js/servicios/consultar.php",
                data: {
                    funcion:"consultarGestionPasajeroEspecifico",
                    aeronave: vm.txtAeronave,
                    pasajero: vm.txtPasajero,
                    estado:1
                }
            }
            var promesa = GestorServicio.llamarFuncionPHP(datos);
                promesa.then(function (respuesta) {
                    vm.existe = respuesta.data.length;
                    vm.tooltipPasajeroNo = false;
                    if(parseInt(vm.existe) === 0){
                        validacionCapacidad();
                    }else{
                        vm.tooltipPasajeroNo = true;
                    }
                },
                function() {
                    agregarAlerta("danger","Error al consultar");
                });
        }
 
        function validacionCapacidad(){
            var datos= {
                metodo: "POST",
                url:"recursos/js/servicios/consultar.php",
                data: {
                    funcion:"consultarGestionPasajeroAeronave",
                    aeronave: vm.txtAeronave,
                    pasajero: vm.txtPasajero,
                    estado:"1"
                }
            }
            var promesa = GestorServicio.llamarFuncionPHP(datos);
                promesa.then(function (respuesta) {
                    vm.totalPasajeros = respuesta.data.length;
                    if(parseInt(vm.totalPasajeros) < parseInt(vm.capacidad)){
                        insertarGestion();
                    }else{
                        agregarAlerta("warning","La Aeronave no tiene mas capacidad. Maxima Capacidad es:" + vm.capacidad);
                    }
                },
                function() {
                    agregarAlerta("danger","Error al consultar");
                });
        }
 
        function consultarGestion(){
            var datos= {
                metodo: "POST",
                url:"recursos/js/servicios/consultar.php",
                data: {
                    funcion:"consultarGestionPasajeroAeronave",
                    aeronave: vm.txtAeronave,
                    pasajero: vm.txtPasajero,
                    estado:"1"
                }
            }
            var promesa = GestorServicio.llamarFuncionPHP(datos);
                promesa.then(function (respuesta) {
                    borrarFiltros();
                    agregarAlerta("success","Datos insertados correctamente");
                    vm.informacionGestion = respuesta.data;
                    vm.dtAdministrar.reloadData();
                },
                function() {
                    agregarAlerta("danger","Error al consultar");
                });
        }
 
        function insertarGestion(){
            var datos= {
                metodo: "POST",
                url:"recursos/js/servicios/consultar.php",
                data: {
                    funcion:"insertarGestion",
                    aeronave: vm.txtAeronave,
                    pasajero: vm.txtPasajero,
                    estado:1
                }
            }
            var promesa = GestorServicio.llamarFuncionPHP(datos);
                promesa.then(function (respuesta) {
 
                    consultarGestion();
                },
                function(error) {
                    agregarAlerta("danger","Error al consultar");
                });
        }
 
        function editarGestion(){
            var datos= {
                metodo: "POST",
                url:"recursos/js/servicios/consultar.php",
                data: {
                    funcion:"modificarGestion",
                    aeronave: vm.txtAeronave,
                    pasajero: vm.txtPasajero,
                    estado:1
                }
            }
            var promesa = GestorServicio.llamarFuncionPHP(datos);
                promesa.then(function (respuesta) {
                    agregarAlerta("success","Datos actualizados correctamente");
                    consultarGestion();
                },
                function() {
                    agregarAlerta("danger","Error al actualizar");
                });
        }
 
        /**
         * @author Mauro Xavier Rivera Rasury (mjrr210316)
         * @description Funcion para agregar alertas a mostrar
         * @param {string} [tipoMensaje] [Definicion del tipo de alerta]
         * @param {string} [mensaje] [Definicion del cuerpo de la alerta]
         * @returns void
         */
        function agregarAlerta(tipoMensaje, mensaje) {
            vm.alerts.push({ type: tipoMensaje, msg: mensaje });
        }
 
 		/**
         * @author Mauro Xavier Rivera Rasury (mjrr210316)
         * @description Funcion para borrar formulario
         * @returns void
         */
        function borrarFiltros() {
            vm.alerts = [];
            vm.tooltipAeronave = false;
            vm.tooltipPasajero = false;
            vm.tooltipPasajeroNo = false;
            vm.txtAeronave = "";
            vm.txtPasajero = "";
            vm.desactivarBtnAceptar = true;
            vm.existeAeronave = true;
            vm.existePasajero = true;
        }
	}
