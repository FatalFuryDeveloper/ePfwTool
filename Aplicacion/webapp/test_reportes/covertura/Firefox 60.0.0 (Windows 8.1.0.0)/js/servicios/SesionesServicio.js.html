/**
 * Tipo: Servicio
 * Nombre: SesionesServicio
 * Descripción: Es el servicio generico que realiza las peticiones http
 * @autor Mauro Xavier Rivera Rasury (mjrr210316)
 * @copyright SRI 2016
 * @param $http - Ambito para acciones http
 * @param $q - Ambito de promesas
 * Copyright 2016 Servicio de Rentas Internas.
 * Todos los derechos reservados.
 */
(function(angular){
    'use strict';
 
    angular.module('sri').service('SesionesServicio', SesionesServicio);
 
    SesionesServicio.$inject = ['$q'];
 
    function SesionesServicio($q) {
        var vm = this ;
 
        /**
         * @description Funcion para realizar peticiones HTTP
         */
        vm.get = function (key) {
            return sessionStorage.getItem(key);
        };
 
        vm.set = function (key,val) {
            return sessionStorage.setItem(key,val);
        };
 
        vm.getList = function (key) {
            return JSON.parse(sessionStorage.getItem(key));
        };
 
        vm.setList = function (key,list) {
             return sessionStorage.setItem(key, JSON.stringify(list));
        };
 
        vm.unset = function (key) {
            return sessionStorage.removeItem(key);
        };
 
        vm.clear = function (key) {
            return sessionStorage.clear();
        };
    }
}(window.angular));
